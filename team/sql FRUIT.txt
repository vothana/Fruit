SELECT
FRUIT.FRUITID,
FRUIT.NAME,
FRUIT.PRICE,
FRUIT.IMAGE,
FRUIT.DISCRIPTION,
CART.QUANTITY
FROM ((ORDERS
INNER JOIN CART
ON ORDERS.ORDERID = CART.ORDERID)
INNER JOIN FRUIT
ON CART.FRUITID = FRUIT.FRUITID)
WHERE ORDERS.STATUS = 'PROCESSING' AND ORDERS.USERID = 2;

ORDERS.STATUS = 'PROCESSING'
